三步驟快速啟用教學

1. 準備 Google 表單

表單包含：桌號（下拉選單 1 至 22 或動態桌號）、作品名稱、構想介紹、與實作規劃檔案（上傳格式限 PDF 或 PPT）。接著在回覆分頁中點擊「連結至試算表」。

2. 貼入 Google Apps Script

在試算表上方點選「延伸功能」->「Apps Script」，清空預設程式，並貼入下方程式碼儲存。

backend.gs (已支援桌號動態雲端同步！) 
/* GAS Backend - Copy to Apps Script */
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var formSheet = ss.getSheets()[0]; 
  var formValues = formSheet.getDataRange().getValues();
  
  // 建立或讀取桌號設定表，實現桌號清單多機動態同步
  var configSheet = ss.getSheetByName("桌號設定");
  if (!configSheet) {
    configSheet = ss.insertSheet("桌號設定");
    configSheet.appendRow(["桌號"]);
    for (var i = 1; i <= 22; i++) {
      configSheet.appendRow([i.toString()]);
    }
  }
  var configValues = configSheet.getDataRange().getValues();
  var activeTables = [];
  for (var c = 1; c < configValues.length; c++) {
    var val = configValues[c][0].toString().trim();
    if (val) activeTables.push(val);
  }

  // 讀取評選狀態
  var statusSheet = ss.getSheetByName("評選狀態");
  if (!statusSheet) {
    statusSheet = ss.insertSheet("評選狀態");
    statusSheet.appendRow(["桌號", "確認狀態", "評審狀態"]);
  }
  var statusValues = statusSheet.getDataRange().getValues();
  
  var tableStates = {};
  activeTables.forEach(function(t) {
    tableStates[t] = { tableNum: t, confirmed: "no", score: "none" };
  });

  for (var j = 1; j < statusValues.length; j++) {
    var tNum = statusValues[j][0].toString().trim();
    if (tableStates[tNum]) {
      tableStates[tNum].confirmed = statusValues[j][1] || "no";
      tableStates[tNum].score = statusValues[j][2] || "none";
    }
  }

  // 分析表單回應資料
  var headers = formValues[0];
  var timestampIdx = 0;
  var tableIdx = 1;
  var nameIdx = 2;
  var descIdx = 3;
  var fileIdx = 4;
  
  for (var h = 0; h < headers.length; h++) {
    var headerText = headers[h].toString();
    if (headerText.indexOf("桌號") > -1) tableIdx = h;
    else if (headerText.indexOf("名稱") > -1 || headerText.indexOf("組別") > -1 || headerText.indexOf("作品") > -1) nameIdx = h;
    else if (headerText.indexOf("構想") > -1 || headerText.indexOf("介紹") > -1) descIdx = h;
    else if (headerText.indexOf("檔案") > -1 || headerText.indexOf("PDF") > -1 || headerText.indexOf("PPT") > -1 || headerText.indexOf("規劃") > -1) fileIdx = h;
  }

  // 過濾學生重複提交，只抓最新一筆
  var latestSubmissions = {};
  for (var k = 1; k < formValues.length; k++) {
    var row = formValues[k];
    var tableNum = row[tableIdx].toString().trim();
    if (tableNum) {
      latestSubmissions[tableNum] = {
        tableNum: tableNum,
        timestamp: row[timestampIdx],
        studentName: row[nameIdx] || "未填寫作品名稱",
        description: row[descIdx] || "無介紹大綱",
        fileUrl: row[fileIdx] || ""
      };
    }
  }

  var finalData = [];
  activeTables.forEach(function(t) {
    var sub = latestSubmissions[t] || null;
    finalData.push({
      tableNum: t,
      hasSubmitted: sub !== null,
      studentName: sub ? sub.studentName : "",
      description: sub ? sub.description : "",
      fileUrl: sub ? sub.fileUrl : "",
      timestamp: sub ? sub.timestamp : "",
      confirmed: tableStates[t].confirmed === "yes",
      score: tableStates[t].score
    });
  });

  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    activeTables: activeTables,
    data: finalData
  })).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var params = JSON.parse(e.postData.contents);
    var action = params.action;
    
    if (action === "updateState") {
      var statusSheet = ss.getSheetByName("評選狀態") || ss.insertSheet("評選狀態");
      var tableNum = params.tableNum.toString();
      var confirmed = params.confirmed;
      var score = params.score;
      var dataRange = statusSheet.getDataRange();
      var values = dataRange.getValues();
      var found = false;
      for (var j = 1; j < values.length; j++) {
        if (values[j][0].toString() === tableNum) {
          if (confirmed !== undefined) statusSheet.getRange(j + 1, 2).setValue(confirmed);
          if (score !== undefined) statusSheet.getRange(j + 1, 3).setValue(score);
          found = true;
          break;
        }
      }
      if (!found) {
        statusSheet.appendRow([tableNum, confirmed || "no", score || "none"]);
      }
    } else if (action === "syncTables") {
      // 同步動態桌號設定
      var configSheet = ss.getSheetByName("桌號設定") || ss.insertSheet("桌號設定");
      configSheet.clear();
      configSheet.appendRow(["桌號"]);
      var tables = params.activeTables;
      for (var i = 0; i < tables.length; i++) {
        configSheet.appendRow([tables[i].toString()]);
      }
    } else if (action === "clearAll") {
      var statusSheet = ss.getSheetByName("評選狀態");
      if (statusSheet) {
        var dataRange = statusSheet.getDataRange();
        var values = dataRange.getValues();
        for (var j = 1; j < values.length; j++) {
          statusSheet.getRange(j + 1, 2).setValue("no");
          statusSheet.getRange(j + 1, 3).setValue("none");
        }
      }
    }
    return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}
